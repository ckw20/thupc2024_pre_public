\documentclass{standalone}

\usepackage{tikz}
\usetikzlibrary{graphs,graphdrawing,arrows}
\usetikzlibrary{automata,
                positioning,
                quotes}

\begin{document}
% https://tex.stackexchange.com/questions/170707/labeling-edges-in-tikz
\begin{tikzpicture}[
node distance = 10mm, on grid,
every edge/.append style = {-},
every edge quotes/.append style = {auto, font=\footnotesize, inner sep=2pt}]
\begin{scope}[nodes={circle, draw, semithick}]
	\node (1)       {1};
	\node (2) [right=of 1] {2};
	\node (3) [right=of 2] {3};
	\node (5) [below=of 3] {5};
	\node (6) [right=of 3] {6};
	\node (7) [above=of 3] {7};
\end{scope}
\begin{scope}[nodes={circle, draw, semithick, fill, text=white}]
	\node (4) [below=of 2] {4};
	\node (8) [right=of 6] {8};
	\node (9) [below=of 8] {9};
	\node (10) [left=of 4, inner sep=2pt] {10};
\end{scope}
%
\path (1)   edge ["4"] (2)
  (2)   edge ["0"] (3)
  (3)   edge ["0"] (6) 
  (6)   edge ["0"] (8)
  (8)   edge ["2"] (9);
% 
\path (2) edge ["0"] (4)
  (4) edge ["2"] (10);
%
\path (7) edge ["4"] (3)
  (3) edge ["4"] (5);
\end{tikzpicture}
\end{document}
